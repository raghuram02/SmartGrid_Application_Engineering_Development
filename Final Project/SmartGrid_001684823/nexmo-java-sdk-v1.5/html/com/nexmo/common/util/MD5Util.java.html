24import java.security.MessageDigest;
25import java.security.NoSuchAlgorithmException;
26import java.io.UnsupportedEncodingException;
27
28/**
29 * MD5Util.java<br><br>
30 *
31 * Contains utility methods that use MD5 hashing. The class uses STANDARD JVM MD5 algorithm<br><br>
32 *
33 * Created on 5 January 2011, 17:34
34 *
35 * @author  Paul Cook
36 * @version 1.0
37 */
38public class MD5Util {
39
40    /**
41     * Calculates MD5 hash for string. assume string is UTF-8 encoded
42     * @param input string which is going to be encoded into MD5 format
43     * @return  MD5 representation of the input string
44     * @throws NoSuchAlgorithmException
45     */
46    public static String calculateMd5(String input) throws NoSuchAlgorithmException {
47        try {
48            return calculateMd5(input, "UTF-8");
49        } catch (UnsupportedEncodingException e) {
50            return null; // -- impossible --
51        }
52    }
53
54    /**
55     * Calculates MD5 hash for string.
56     * @param input string which is going to be encoded into MD5 format
57     * @param encoding character encoding of the string which is going to be encoded into MD5 format
58     * @return  MD5 representation of the input string
59     * @throws NoSuchAlgorithmException
60     */
61    public static String calculateMd5(String input, String encoding) throws NoSuchAlgorithmException, UnsupportedEncodingException {
62        MessageDigest md = MessageDigest.getInstance("MD5");
63        md.update(input.getBytes(encoding));
64        byte digest[] = md.digest();
65
66        final StringBuilder hexString = new StringBuilder();
67        for (byte element : digest) {
68            int z = 0xFF & element;
69            if (z < 16)
70                hexString.append("0");
71            hexString.append(Integer.toHexString(z));
72        }
73
74        return hexString.toString();
75    }
76
77}
