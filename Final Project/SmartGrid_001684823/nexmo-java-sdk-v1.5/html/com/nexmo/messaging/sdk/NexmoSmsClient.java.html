58/**
59 * NexmoSmsClient.java<br><br>
60 *
61 * Client for talking to the Nexmo REST interface<br><br>
62 *
63 * Usage
64 *
65 * To submit a message, first you should instanciate a NexmoSmsClient, passing the credentials for your Nexmo account on the constructor.
66 * Then, you should instanciate the appropriate {@link com.nexmo.messaging.sdk.messages.Message} subclass depending on which type of message you are going to submit.<br>
67 * The following subclasses are available ...<br><br>
68 *
69 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.TextMessage}<br>
70 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.BinaryMessage}<br>
71 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.WapPushMessage}<br>
72 * com.nexmo.messaging.sdk.messages.{@link com.nexmo.messaging.sdk.messages.UnicodeMessage}<br>
73 *
74 * Each of these subclasses requires different message parameters to be passed on the constructor.
75 * See the included javadocs for further details.<br><br>
76 *
77 * Once you have a {@link com.nexmo.messaging.sdk.messages.Message} object, you simply pass this to the {@link #submitMessage(Message)} method in the NexmoSmsClient instance.
78 * This will construct and post the request to the Nexmo REST service.<br>
79 * This method will return an array of {@link com.nexmo.messaging.sdk.SmsSubmissionResult}, with 1 entry for every sms message that was sent.<br>
80 * Certain messages, for example, long text messages greater than 160 characters, will require multiple sms's to be submitted.<br>
81 * Each entry in this array will contain an individual messageId as well as an individual status detailing the success or reason for failure of each message.<br><br>
82 *
83 * The list of possible status codes is listed below ..<br><br>
84 *
85 * // 0 = success  -- the message was successfully accepted for delivery by nexmo<br>
86 * // 1 = throttled -- You have exceeded the submission capacity allowed on this account, please back-off and re-try<br>
87 * // 2 = missing params -- Your request is incomplete and missing some mandatory parameters<br>
88 * // 3 = invalid params -- The value of 1 or more parameters is invalid<br>
89 * // 4 = invalid credentials -- The api key / secret you supplied is either invalid or disabled<br>
90 * // 5 = internal error -- An error has occurred in the nexmo platform whilst processing this message<br>
91 * // 6 = invalid message -- The Nexmo platform was unable to process this message, for example, an un-recognized number prefix<br>
92 * // 7 = number barred -- The number you are trying to submit to is blacklisted and may not receive messages<br>
93 * // 8 = partner account barred -- The api key you supplied is for an account that has been barred from submitting messages<br>
94 * // 9 = partner quota exceeded -- Your pre-pay account does not have sufficient credit to process this message<br>
95 * // 10 = too many existing binds -- The number of simultaneous connections to the platform exceeds the capabilities of your account<br>
96 * // 11 = account not enabled for http -- This account is not provisioned for http / rest submission, you should use SMPP instead<br>
97 * // 12 = message too long -- The message length exceeds the maximum allowed<br>
98 * // 13 = comms failure -- There was a network failure attempting to contact Nexmo
99 * // 14 = Invalid Signature -- The signature supplied with this request was not verified successfully
00 * // 15 = invalid sender address -- The sender address was not allowed for this message
01 * // 16 = invalid TTL -- The ttl parameter values, or combination of parameters is invalid
02 * // 17 = number unreachable -- this destination cannot be delivered to at this time (if reachable=true is specified)
03 * // 18 = too many destinations -- There are more than the maximum allowed number of destinations in this request
04 * // 19 = Facility Not Allowed - Your request makes use of a facility that is not enabled on your account
05 * // 20 = Invalid Message Class - The message class value supplied was out of range (0 - 3)
06 *
07 * @author  Paul Cook
08 * @version 1.0
09 */

04     * Instanciate a new NexmoSmsClient instance that will communicate using the supplied credentials, and will use the supplied connection and read timeout values.<br>
05     * Additionally, you can specify an alternative service base url. For example submitting to a testing sandbox environment,
06     * or if requested to submit to an alternative address by Nexmo, for example, in cases where it may be necessary to prioritize your traffic.
07     *
08     * @param apiKey Your Nexmo account api key
09     * @param apiSecret Your Nexmo account api secret
10     * @param connectionTimeout over-ride the default connection timeout with this value (in milliseconds)
11     * @param soTimeout over-ride the default read-timeout with this value (in milliseconds)
12     * @param signRequests do we generate a signature for this request using the secret key
13     * @param signatureSecretKey the secret key we will use to generate the signatures for signed requests
14     * @param useSSL do we use a SSL / HTTPS connection for submitting requests
15     */
16    public NexmoSmsClient(String baseUrl,
17                          final String apiKey,
18                          final String apiSecret,
19                          final int connectionTimeout,
20                          final int soTimeout,
21                          final boolean signRequests,
22                          final String signatureSecretKey,
23                          final boolean useSSL) throws Exception {
24
25        // Derive a http and a https version of the supplied base url
26        baseUrl = baseUrl.trim();
27        String lc = baseUrl.toLowerCase();
28        if (!lc.startsWith("http://") && !lc.startsWith("https://"))
29            throw new Exception("base url does not start with http:// or https://");
30        if (lc.startsWith("http://")) {
31            this.baseUrlHttp = baseUrl;
32            this.baseUrlHttps = "https://" + baseUrl.substring(7);
33        } else {
34            this.baseUrlHttps = baseUrl;
35            this.baseUrlHttp = "http://" + baseUrl.substring(8);
36        }
37
38        this.apiKey = apiKey;
39        this.apiSecret = apiSecret;
40        this.connectionTimeout = connectionTimeout;
41        this.soTimeout = soTimeout;
42        try {
43            this.documentBuilderFactory = DocumentBuilderFactory.newInstance();
44            this.documentBuilder = this.documentBuilderFactory.newDocumentBuilder();
45        } catch (javax.xml.parsers.ParserConfigurationException e) {
46            throw new Exception("ERROR initializing XML Document builder!", e);
47        }
48
49        this.signRequests = signRequests;
50        this.signatureSecretKey = signatureSecretKey;
51
52        this.useSSL = useSSL;
53    }

68     *

70     */

72        boolean performReachabilityCheck = false;
73        ValidityPeriod validityPeriod = null;
74        String networkCode = null;

02     * submit a message submission request object.
03     * This will use the supplied object to construct a request and post it to the Nexmo REST interface.<br>
04     * This method will respond with an array of SmsSubmissionResult objects. Depending on the nature and length of the submitted message, Nexmo may automatically
05     * split the message into multiple sms messages in order to deliver to the handset. For example, a long text sms of greater than 160 chars will need to be split
06     * into multiple 'concatenated' sms messages. The Nexmo service will handle this automatically for you.<br>
07     * The array of SmsSubmissionResult objects will contain a SmsSubmissionResult object for every actual sms that was required to submit the message.
08     * each message can potentially have a different status result, and each message will have a different message id.
09     * Delivery notifications will be generated for each sms message within this set and will be posted to your application containing the appropriate message id.
10     *
11     * @param message The message request object that describes the type of message and the contents to be submitted.
12     * @param validityPeriod The validity period (Time-To-Live) for this message. Specifies the time before this mesage will be expired if not yet delivered
13     * @param networkCode (Optional) use this parameter to force this message to be associated with and delivered on this network. Use this in cases where you want to over-ride
14     *                               the automatic network detection provided by Nexmo. This value will be used in order to determine the pricing and routing for this message.<br>
15     *                               (Note) This feature must be enabled and available on your account or else this value will be ignored.
16     * @param performReachabilityCheck Flag to indicate wether a reachability check should be performed on this message before delivery is attempted. If the destination is
17     *                                 not reachable, the message will be rejected and a reachability status will be returned in the response field smsSubmissionReachabilityStatus<br>
18     *                                 (Note) This feature must be enabled and available on your account or else the message request will be rejected. There may be additional cost
19     *                                 associated with the use of this feature.
20     *
21     * @return SmsSubmissionResult[] an array of results, 1 object for each sms message that was required to submit this message in its entirety
22     *
23     * @throws Exception There has been a general failure either within the Client class, or whilst attempting to communicate with the Nexmo service (eg, Network failure)
24     */
25    public SmsSubmissionResult[] submitMessage(Message message,
26                                               ValidityPeriod validityPeriod,
27                                               String networkCode,
28                                               boolean performReachabilityCheck) throws Exception {
29
30        log.debug("HTTP-Message-Submission Client .. from [ " + message.getFrom() + " ] to [ " + message.getTo() + " ] msg [ " + message.getMessageBody() + " ] ");
31
32        // From the Message object supplied, construct an appropriate request to be submitted to the Nexmo REST Service.
33
34        // Determine what 'product' type we are submitting, and select the appropriate endpoint path
35        String submitPath = SUBMISSION_PATH_SMS;
36        if (message.getType() == Message.MESSAGE_TYPE_USSD_DISPLAY)
37            submitPath = SUBMISSION_PATH_USSD_DISPLAY;
38        if (message.getType() == Message.MESSAGE_TYPE_USSD_PROMPT)
39            submitPath = SUBMISSION_PATH_USSD_PROMPT;
40
41        // Determine the type parameter based on the type of Message object.
42        boolean binary = message.getType() == Message.MESSAGE_TYPE_BINARY;
43        boolean unicode = message.isUnicode();
44        boolean wapPush = message.getType() == Message.MESSAGE_TYPE_WAPPUSH;
45
46        String mode = "text";
47
48        if (binary)
49            mode = "binary";
50
51        if (unicode)
52            mode = "unicode";
53
54        if (wapPush)
55            mode = "wappush";
56
57        // Construct a query string as a list of NameValuePairs
58
59        List<NameValuePair> params = new ArrayList<NameValuePair>();
60
61        boolean doPost = false;
62
63        params.add(new BasicNameValuePair("api_key", this.apiKey));
64        if (!this.signRequests)
65            params.add(new BasicNameValuePair("api_secret", this.apiSecret));
66        params.add(new BasicNameValuePair("from", message.getFrom()));
67        params.add(new BasicNameValuePair("to", message.getTo()));
68        params.add(new BasicNameValuePair("type", mode));
69        if (wapPush) {
70            params.add(new BasicNameValuePair("url", message.getWapPushUrl()));
71            params.add(new BasicNameValuePair("title", message.getWapPushTitle()));
72            if (message.getWapPushValidity() > 0)
73                params.add(new BasicNameValuePair("validity", "" + message.getWapPushValidity()));
74        } else if (binary) {
75            // Binary Message
76            if (message.getBinaryMessageUdh() != null)
77                params.add(new BasicNameValuePair("udh", HexUtil.bytesToHex(message.getBinaryMessageUdh())));
78            params.add(new BasicNameValuePair("body", HexUtil.bytesToHex(message.getBinaryMessageBody())));
79        } else {
80            // Text Message
81            params.add(new BasicNameValuePair("text", message.getMessageBody()));
82            if (message.getMessageBody() != null && message.getMessageBody().length() > 255)
83                doPost = true;
84        }
85
86        if (message.getClientReference() != null)
87            params.add(new BasicNameValuePair("client-ref", message.getClientReference()));
88
89        params.add(new BasicNameValuePair("status-report-req", "" + message.getStatusReportRequired()));
90
91        if (message.getMessageClass() != null)
92            params.add(new BasicNameValuePair("message-class", "" + message.getMessageClass().getMessageClass()));
93
94        if (message.getProtocolId() != null)
95            params.add(new BasicNameValuePair("protocol-id", "" + message.getProtocolId()));
96
97        if (validityPeriod != null) {
98            if (validityPeriod.getTimeToLive() != null)
99                params.add(new BasicNameValuePair("ttl", "" + validityPeriod.getTimeToLive().intValue()));
00            if (validityPeriod.getValidityPeriodHours() != null)
01                params.add(new BasicNameValuePair("ttl-hours", "" + validityPeriod.getValidityPeriodHours().intValue()));
02            if (validityPeriod.getValidityPeriodMinutes() != null)
03                params.add(new BasicNameValuePair("ttl-minutes", "" + validityPeriod.getValidityPeriodMinutes().intValue()));
04            if (validityPeriod.getValidityPeriodSeconds() != null)
05                params.add(new BasicNameValuePair("ttl-seconds", "" + validityPeriod.getValidityPeriodSeconds().intValue()));
06        }
07
08        if (networkCode != null)
09            params.add(new BasicNameValuePair("network-code", networkCode));
10
11        if (performReachabilityCheck)
12            params.add(new BasicNameValuePair("test-reachable", "true"));
13
14        if (this.signRequests)
15            RequestSigning.constructSignatureForRequestParameters(params, this.signatureSecretKey);
16
17        String baseUrl = this.useSSL ? this.baseUrlHttps : this.baseUrlHttp;
18        baseUrl = baseUrl + submitPath;
19
20        // Now that we have generated a query string, we can instanciate a HttpClient,
21        // construct a POST or GET method and execute to submit the request
22        String response = null;
23        for (int pass=1;pass<=2;pass++) {
24            HttpUriRequest method = null;
25            doPost = true;
26            String url = null;
27            if (doPost) {
28                HttpPost httpPost = new HttpPost(baseUrl);
29                httpPost.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
30                method = httpPost;
31                url = baseUrl + "?" + URLEncodedUtils.format(params, "utf-8");
32            } else {
33                String query = URLEncodedUtils.format(params, "utf-8");
34                method = new HttpGet(baseUrl + "?" + query);
35                url = method.getRequestLine().getUri();
36            }
37
38            try {
39                if (this.httpClient == null)
40                    this.httpClient = HttpClientUtils.getInstance(this.connectionTimeout, this.soTimeout).getNewHttpClient();
41                HttpResponse httpResponse = this.httpClient.execute(method);
42                int status = httpResponse.getStatusLine().getStatusCode();
43                if (status != 200)
44                    throw new Exception("got a non-200 response [ " + status + " ] from Nexmo-HTTP for url [ " + url + " ] ");
45                response = new BasicResponseHandler().handleResponse(httpResponse);
46                log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
47                break;
48            } catch (Exception e) {
49                method.abort();
50                log.info("communication failure: " + e);
51                String exceptionMsg = e.getMessage();
52                if (exceptionMsg.indexOf("Read timed out") >= 0) {
53                    log.info("we're still connected, but the target did not respond in a timely manner ..  drop ...");
54                } else {
55                    if (pass == 1) {
56                        log.info("... re-establish http client ...");
57                        this.httpClient = null;
58                        continue;
59                    }
60                }
61
62                // return a COMMS failure ...
63                SmsSubmissionResult[] results = new SmsSubmissionResult[1];
64                results[0] = new SmsSubmissionResult(SmsSubmissionResult.STATUS_COMMS_FAILURE,
65                                                     null,
66                                                     null,
67                                                     "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
68                                                     message.getClientReference(),
69                                                     null,
70                                                     null,
71                                                     true,
72                                                     null,
73                                                     null);
74                return results;
75            }
76        }
77
78        // parse the response doc ...
79
80        /*
81            We receive a response from the api that looks like this, parse the document
82            and turn it into an array of SmsSubmissionResult, one object per <message> node
83
84                <mt-submission-response>
85                    <messages count='x'>
86                        <message>
87                            <to>xxx</to>
88                            <messageId>xxx</messageId>
89                            <status>xx</status>
90                            <errorText>ff</errorText>
91                            <clientRef>xxx</clientRef>
92                            <remainingBalance>##.##</remainingBalance>
93                            <messagePrice>##.##</messagePrice>
94                            <reachability status='x' description='xxx' />
95                            <network>23410</network>
96                        </message>
97                    </messages>
98                </mt-submission-response>
99        */
00
01        List<SmsSubmissionResult> results = new ArrayList<SmsSubmissionResult>();
02
03        Document doc = null;
04        synchronized(this.documentBuilder) {
05            try {
06                doc = this.documentBuilder.parse(new InputSource(new StringReader(response)));
07            } catch (Exception e) {
08                throw new Exception("Failed to build a DOM doc for the xml document [ " + response + " ] ", e);
09            }
10        }
11
12        NodeList replies = doc.getElementsByTagName("mt-submission-response");
13        for (int i=0;i<replies.getLength();i++) {
14            Node reply = replies.item(i);
15            NodeList messageLists = reply.getChildNodes();
16            for (int i2=0;i2<messageLists.getLength();i2++) {
17                Node messagesNode = messageLists.item(i2);
18                if (messagesNode.getNodeType() != Node.ELEMENT_NODE)
19                    continue;
20                if (messagesNode.getNodeName().equals("messages")) {
21                    NodeList messages = messagesNode.getChildNodes();
22                    for (int i3=0;i3<messages.getLength();i3++) {
23                        Node messageNode = messages.item(i3);
24                        if (messageNode.getNodeType() != Node.ELEMENT_NODE)
25                            continue;
26
27                        int status = -1;
28                        String messageId = null;
29                        String destination = null;
30                        String errorText = null;
31                        String clientReference = null;
32                        BigDecimal remainingBalance = null;
33                        BigDecimal messagePrice = null;
34                        SmsSubmissionReachabilityStatus smsSubmissionReachabilityStatus = null;
35                        String network = null;
36
37                        NodeList nodes = messageNode.getChildNodes();
38                        for (int i4=0;i4<nodes.getLength();i4++) {
39                            Node node = nodes.item(i4);
40                            if (node.getNodeType() != Node.ELEMENT_NODE)
41                                continue;
