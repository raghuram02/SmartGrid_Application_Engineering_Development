24/**
25 * SmsSubmissionReachabilityStatus.java<br><br>
26 *
27 * Created on 5 January 2011, 17:34<br><br>
28 *
29 * Represents the result of the reachability check that was performed when submitting the message if such a check was requested
30 *
31 * @author  Paul Cook
32 * @version 1.0
33 */
34public class SmsSubmissionReachabilityStatus implements java.io.Serializable {
35
36    static final long serialVersionUID = 8121497095898864184L;
37
38    /**
39     * The reachability status of this messages destination could not be determined
40     */
41    public static final int REACHABILITY_STATUS_UNKNOWN = 0;
42
43    /**
44     * The destination of this message is reachable
45     */
46    public static final int REACHABILITY_STATUS_REACHABLE = 1;
47
48    /**
49     * The destination of this message can not be reached and thus can not be delivered to
50     */
51    public static final int REACHABILITY_STATUS_UNDELIVERABLE = 2;
52
53    /**
54     * The destinationof this message is temporarily unavailable (eg, switched off)
55     */
56    public static final int REACHABILITY_STATUS_ABSENT = 3;
57
58    /**
59     * The destination of this message is not a valid destination and thus could not be delivered to
60     */
61    public static final int REACHABILITY_STATUS_BAD_NUMBER = 4;
62
63    private final int status;
64    private final String description;
65
66    protected SmsSubmissionReachabilityStatus(final int status,
67                                              final String description) {
68        this.status = status;
69        this.description = description;
70    }
71
72    /**
73     * @return int status code representing outcome of the reachability check performed on this message
74     */
75    public int getStatus() {
76        return this.status;
77    }
78
79    /**
80     * @return String A description of the outcome of the reachability check performed on this message
81     */
82    public String getDescription() {
83        return this.description;
84    }
85
86    @Override
87    public String toString() {
88        return "REACHABILITY -- STAT [ " + this.status + " ] [ " + this.description + " ] ";
89    }
90
91}
