24import java.math.BigDecimal;
25
26/**
27 * SmsSubmissionResult.java<br><br>
28 *
29 * Created on 5 January 2011, 17:34<br><br>
30 *
31 * Represents the result of the submission of a single sms message,
32 * or a single part of a long / multi-part message
33 *
34 * @author  Paul Cook
35 * @version 1.0
36 */
37public class SmsSubmissionResult implements java.io.Serializable {
38
39    static final long serialVersionUID = 2580996244288340269L;
40
41    /**
42     * Message was successfully submitted to the Nexmo service
43     */
44    public static final int STATUS_OK = 0;
45
46    /**
47     * Message was rejected due to exceeding the maximum throughput allowed for this account.<br>
48     * Message can be re-submitted after a short delay
49     */
50    public static final int STATUS_THROTTLED = 1;
51
52    /**
53     * Message was rejected due to incomplete data in the submission request
54     */
55    public static final int STATUS_MISSING_PARAMS = 2;
56
57    /**
58     * Message was rejected due to an illegal value in one or more elements of the submission request
59     */
60    public static final int STATUS_INVALID_PARAMS = 3;
61
62    /**
63     * Message was rejected due to receiving invalid account api key and/or secret
64     */
65    public static final int STATUS_INVALID_CREDENTIALS = 4;
66
67    /**
68     * Message was rejected due to a failure within the Nexmo systems.<br>
69     * Message can be re-submitted after a short delay
70     */
71    public static final int STATUS_INTERNAL_ERROR = 5;
72
73    /**
74     * Message was rejected because the Nexmo service was unable to handle this request. eg, the destination was un-routable.
75     */
76    public static final int STATUS_INVALID_MESSAGE = 6;
77
78    /**
79     * Message was rejected because the phone number you tried to submit to has been blacklisted.
80     */
81    public static final int STATUS_NUMBER_BARRED = 7;
82
83    /**
84     * Message was rejected because your account has been barred, or has not yet been activated
85     */
86    public static final int STATUS_PARTNER_ACCOUNT_BARRED = 8;
87
88    /**
89     * Message was rejected because your pre-paid balance does not contain enough credit to handle this request.<br>
90     * Please top up your balance before re-submitting this request or subsequent requests.
91     */
92    public static final int STATUS_PARTNER_QUOTA_EXCEEDED = 9;
93
94    /**
95     * Message was rejected because already have the maximum number of concurrent connections allowed for your account.
96     */
97    public static final int STATUS_TOO_MANY_BINDS = 10;
98
99    /**
00     * Message was rejected because your account is not provisioned for submitting via the REST interface.
01     */
02    public static final int STATUS_ACCOUNT_NOT_HTTP = 11;
03
04    /**
05     * Message was rejected because it exceeds the allowable maximum length (140 octets for a binary message, or 3200 chars for a text message)
06     */
07    public static final int STATUS_MESSAGE_TOO_LONG = 12;
08
09    /**
10     * Message was not submitted because there was a communications failure.
11     */
12    public static final int STATUS_COMMS_FAILURE = 13;
13
14    /**
15     * Message was not submitted due to a verification failure in the submitted signature
16     */
17    public static final int STATUS_INVALID_SIGNATURE = 14;
18
19    /**
20     * The 'from' address specified in the message submission was not permitted
21     */
22    public static final int STATUS_INVALID_FROM_ADDRESS = 15;
23
24    /**
25     * invalid TTL -- The ttl parameter values, or combination of parameters is invalid
26     */
27    public static final int STATUS_INVALID_TTL = 16;
28
29    /**
30     * This destination cannot be delivered to at this time (if reachable=true is specified)
31     */
32    public static final int STATUS_NUMBER_UNREACHABKE = 17;
33
34    /**
35     * There are more than the maximum allowed number of destinations in this request
36     */
37    public static final int STATUS_TOO_MANY_DESTINATIONS = 18;
38
39    /**
40     * Your request makes use of a facility that is not enabled on your account
41     */
42    public static final int STATUS_FACILITY_NOT_ALLOWED = 19;
43
44    /**
45     * The message class value supplied was out of range (0 - 3)
46     */
47    public static final int STATUS_INVALID_MESSAFE_CLASS = 20;
48
49
50    private final int status;
51    private final String destination;
52    private final String messageId;
53    private final String errorText;
54    private final String clientReference;
55    private final BigDecimal remainingBalance;
56    private final BigDecimal messagePrice;
57    private final boolean temporaryError;
58    private final SmsSubmissionReachabilityStatus smsSubmissionReachabilityStatus;
59    private final String network;
60
61    protected SmsSubmissionResult(final int status,
62                                  final String destination,
63                                  final String messageId,
64                                  final String errorText,
65                                  final String clientReference,
66                                  final BigDecimal remainingBalance,
67                                  final BigDecimal messagePrice,
68                                  final boolean temporaryError,
69                                  final SmsSubmissionReachabilityStatus smsSubmissionReachabilityStatus,
70                                  final String network) {
71        this.status = status;
72        this.destination = destination;
73        this.messageId = messageId;
74        this.errorText = errorText;
75        this.clientReference = clientReference;
76        this.remainingBalance = remainingBalance;
77        this.messagePrice = messagePrice;
78        this.temporaryError = temporaryError;
79        this.smsSubmissionReachabilityStatus = smsSubmissionReachabilityStatus;
80        this.network = network;
81    }
82
83    /**
84     * @return int status code representing either the success of the message submission or a reason for failure
85     */
86    public int getStatus() {
87        return this.status;
88    }
89
90    /**
91     * @return String the destination phone number that the message was submitted to
92     */
93    public String getDestination() {
94        return this.destination;
95    }
96
97    /**
98     * @return String a unique identifier associated with the submitted message.
99     *         This value will be returned in any subsequent delivery notifications in order that they may be correlated with the appropriate message submission request.
00     */
01    public String getMessageId() {
02        return this.messageId;
03    }
04
05    /**
06     * @return String a human readable error message giving further description of the value of {see #getStatus()}
07     */
08    public String getErrorText() {
09        return this.errorText;
10    }
11
12    /**
13     * @return String the client-reference that was supplied as part of the original message submission
14     */
15    public String getClientReference() {
16        return this.clientReference;
17    }
18
19    /**
20     * @return BigDecimal The account balance that remains after the submission of this message.
21     */
22    public BigDecimal getRemainingBalance() {
23        return this.remainingBalance;
24    }
25
26    /**
27     * @return BigDecimal The amount that was debited from your account balance upon submission of this message
28     */
29    public BigDecimal getMessagePrice() {
30        return this.messagePrice;
31    }
32
33    /**
34     * @return boolean Indicates if the failure was due to a temporary condition. If so, then the message submission may be re-attempted after a short delay
35     */
36    public boolean getTemporaryError() {
37        return this.temporaryError;
38    }
39
40    /**
41     * @return SmsSubmissionReachabilityStatus the result of any reachability check that was performed on this message if one was requested
42     */
43    public SmsSubmissionReachabilityStatus getSmsSubmissionReachabilityStatus() {
44        return this.smsSubmissionReachabilityStatus;
45    }
46
47    /**
48     * @return String the 'estimated' network that has been identified for this destination (Note, this can change during the processing of this message due to HLR lookups)
49     */
50    public String getNetwork() {
51        return this.network;
52    }
53
54    @Override
55    public String toString() {
56        StringBuilder sb = new StringBuilder();
57        sb.append("SMS-SUBMIT-RESULT -- STATUS:").append(this.status);
58        sb.append(" ERR:").append(this.errorText);
59        sb.append(" DEST:").append(this.destination);
60        sb.append(" MSG-IG:").append(this.messageId);
61        sb.append(" CLIENT-REF:").append(this.clientReference);
62        sb.append(" PRICE:").append(this.messagePrice == null ? "-" : this.messagePrice.toPlainString());
63        sb.append(" BALANCE:").append(this.remainingBalance == null ? "-" : this.remainingBalance.toPlainString());
64        sb.append(" TEMP-ERR?:").append(this.temporaryError);
65        if (this.smsSubmissionReachabilityStatus != null)
66            sb.append(" REACHABLE?:").append(this.smsSubmissionReachabilityStatus);
67        if (this.network != null)
68            sb.append("NETWORK:").append(this.network);
69
70        return sb.toString();
71    }
