24import com.nexmo.messaging.sdk.NexmoSmsClient;
25import com.nexmo.messaging.sdk.SmsSubmissionResult;
26import com.nexmo.messaging.sdk.messages.WapPushMessage;
27
28/**
29 * SendWapPush.java<br><br>
30 *
31 * An example of using the nexmo sms api to submit a wap push message ...<br><br>
32 *
33 * Created on 5 January 2011, 17:34
34 *
35 * @author  Paul Cook
36 * @version 1.0
37 */
38public class SendWapPush {
39
40    public static final String API_KEY = "your-api-key-goes-here";
41    public static final String API_SECRET = "your-api-secret-goes-here";
42
43    public static final String WAP_PUSH_FROM = "12345";
44    public static final String WAP_PUSH_TO = "447777111222";
45    public static final String WAP_PUSH_URL = "http://www.nexmo.com";
46    public static final String WAP_PUSH_TITLE = "Nexmo";
47
48    public static void main(String[] args) {
49
50        // Create a client for submitting to Nexmo
51
52        NexmoSmsClient client = null;
53        try {
54            client = new NexmoSmsClient(API_KEY, API_SECRET);
55        } catch (Exception e) {
56            System.err.println("Failed to instanciate a Nexmo Client");
57            e.printStackTrace();
58            throw new RuntimeException("Failed to instanciate a Nexmo Client");
59        }
60
61        // Create a Wap-Push Message request object ...
62
63        WapPushMessage message = new WapPushMessage(WAP_PUSH_FROM, WAP_PUSH_TO, WAP_PUSH_URL, WAP_PUSH_TITLE);
64
65        // Use the Nexmo client to submit the Wap Push Message ...
66
67        SmsSubmissionResult[] results = null;
68        try {
69            results = client.submitMessage(message);
70        } catch (Exception e) {
71            System.err.println("Failed to communicate with the Nexmo Client");
72            e.printStackTrace();
73            throw new RuntimeException("Failed to communicate with the Nexmo Client");
74        }
75
76        // Evaluate the results of the submission attempt ...
77        System.out.println("... Message submitted in [ " + results.length + " ] parts");
78        for (int i=0;i<results.length;i++) {
79            System.out.println("--------- part [ " + (i + 1) + " ] ------------");
80            System.out.println("Status [ " + results[i].getStatus() + " ] ...");
81            if (results[i].getStatus() == SmsSubmissionResult.STATUS_OK)
82                System.out.println("SUCCESS");
83            else if (results[i].getTemporaryError())
84                System.out.println("TEMPORARY FAILURE - PLEASE RETRY");
85            else
86                System.out.println("SUBMISSION FAILED!");
87            System.out.println("Message-Id [ " + results[i].getMessageId() + " ] ...");
88            System.out.println("Error-Text [ " + results[i].getErrorText() + " ] ...");
89
90            if (results[i].getMessagePrice() != null)
91                System.out.println("Message-Price [ " + results[i].getMessagePrice() + " ] ...");
92            if (results[i].getRemainingBalance() != null)
93                System.out.println("Remaining-Balance [ " + results[i].getRemainingBalance() + " ] ...");
94        }
95    }
96
97}
