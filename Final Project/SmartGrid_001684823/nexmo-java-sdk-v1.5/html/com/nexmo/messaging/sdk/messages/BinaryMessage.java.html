24import com.nexmo.messaging.sdk.messages.parameters.MessageClass;

26/**
27 * BinaryMessage.java<br><br>
28 *
29 * Represents the details of a binary message that is to be submitted via the Nexmo REST api<br><br>
30 *
31 * Created on 5 January 2011, 17:34
32 *
33 * @author  Paul Cook
34 * @version 1.0
35 */
36public class BinaryMessage extends Message {
37
38    static final long serialVersionUID = -5301775803396025570L;
39
40    /**
41     * Instanciate a new binary sms message request.
42     *
43     * @param from the 'from' address that will be seen on the handset when this message arrives,
44     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
45     * @param to   the phone number of the handset that you wish to send the message to
46     * @param binaryMessageBody The raw binary message data to be sent to a handset.
47     *                          This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
48     *                          so you should ensure that it is a correctly constructed message
49     * @param binaryMessageUdh Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
50     *                         (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
51     *                         Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
52     *                         using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
53     *                         UserDataHeader field that describes the segmentation/re-assembly fields required to sucessfully concatenate multiple short messages.
54     */
55    public BinaryMessage(final String from,
56                         final String to,
57                         final byte[] binaryMessageBody,
58                         final byte[] binaryMessageUdh) {
59        super(MESSAGE_TYPE_BINARY,
60              from,
61              to,
62              null,
63              binaryMessageBody,
64              binaryMessageUdh,
65              null,
66              false,
67              false,
68              null,
69              null,
70              0,
71              null,
72              null);
73    }
74
75    /**
76     * Instanciate a new binary sms message request, attaching an optional client reference, message-class or TTL, and optionally requesting a delivery notification.
77     *
78     * @param from the 'from' address that will be seen on the handset when this message arrives,
79     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
80     * @param to   the phone number of the handset that you wish to send the message to
81     * @param binaryMessageBody The raw binary message data to be sent to a handset.
82     *                          This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
83     *                          so you should ensure that it is a correctly constructed message
84     * @param binaryMessageUdh Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
85     *                         (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
86     *                         Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
87     *                         using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
88     *                         UserDataHeader field that describes the segmentation/re-assembly fields required to sucessfully concatenate multiple short messages.
89     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
90     *                        In order to help with reconcilliation of messages
91     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
92     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
93     *                             Nexmo REST account that represents the outcome of this message.
94     * @param messageClass (Optional) The message class that is to be applied to this message.
95     * @param protocolId The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
96     */
97    public BinaryMessage(final String from,
98                         final String to,
99                         final byte[] binaryMessageBody,
00                         final byte[] binaryMessageUdh,
01                         final String clientReference,
02                         final boolean statusReportRequired,
03                         final MessageClass messageClass,
04                         final Integer protocolId) {
05        super(MESSAGE_TYPE_BINARY,
06              from,
07              to,
08              null,
09              binaryMessageBody,
10              binaryMessageUdh,
11              clientReference,
12              false,
13              statusReportRequired,
14              null,
15              null,
16              0,
17              messageClass,
18              protocolId);
19    }

21}
