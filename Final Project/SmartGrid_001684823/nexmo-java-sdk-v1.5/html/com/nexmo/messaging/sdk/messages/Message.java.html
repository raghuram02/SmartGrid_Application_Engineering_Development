24import com.nexmo.messaging.sdk.messages.parameters.MessageClass;
25
26/**
27 * Message.java<br><br>
28 *
29 * Created on 5 January 2011, 17:34<br><br>
30 *
31 * Represents the details of a message that is to be submitted via the http api.<br>
32 * Not instanciated directly, but instead via one of the sub-classes that represent a particular message type
33 *
34 * @author  Paul Cook
35 * @version 1.0
36 */
37public abstract class Message implements java.io.Serializable {
38
39    private static final long serialVersionUID = 3847700435531116012L;
40
41    /**
42     * Message ia a regular TEXT SMS message
43     */
44    public static final int MESSAGE_TYPE_TEXT = 1;
45    /**
46     * Message ia a binary SMS message with a custom UDH and binary payload
47     */
48    public static final int MESSAGE_TYPE_BINARY = 2;
49    /**
50     * Message is a wap-push message to send a browsable / downloadable url to the handset
51     */
52    public static final int MESSAGE_TYPE_WAPPUSH = 3;
53    /**
54     * Message is a unicode message, for sending messages in non-latin script to a supported handset
55     */
56    public static final int MESSAGE_TYPE_UNICODE = 4;
57
58    /**
59     * Message is a USSD 'display' message that displays on handset but does not require a response
60     */
61    public static final int MESSAGE_TYPE_USSD_DISPLAY = 5;
62
63    /**
64     * Message is a USSD 'prompt' message that requires a response from the end user
65     */
66    public static final int MESSAGE_TYPE_USSD_PROMPT = 6;
67
68    private final int type;
69
70    private final String from;
71    private final String to;
72    private final String messageBody;
73    private final byte[] binaryMessageBody;
74    private final byte[] binaryMessageUdh;
75    private final String clientReference;
76    private final boolean unicode;
77    private final boolean statusReportRequired;
78    private final String wapPushUrl;
79    private final String wapPushTitle;
80    private final int wapPushValidity;
81    private final MessageClass messageClass;
82    private final Integer protocolId;
83
84    /**
85     * Instanciate a new message request.<br>
86     * This constructor exposes the full range of possible parameters and is not for general use
87     * Instead, it is accessed via super() in the constructors of various sub-classes that expose a relevant
88     * sub-set of the available parameters
89     *
90     * @param type the type of message will influence the makeup of the request we post to the Nexmo server, and also the action taken by the Nexmo server in response to this message
91     * @param from the 'from' address that will be seen on the handset when this message arrives,
92     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
93     * @param to   the phone number of the handset that you wish to send the message to
94     * @param messageBody The text of the message to be sent to the handset
95     * @param binaryMessageBody The raw binary message data to be sent to a handset.
96     *                          This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
97     *                          so you should ensure that it is a correctly constructed message
98     * @param binaryMessageUdh Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
99     *                         (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
00     *                         Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
01     *                         using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
02     *                         UserDataHeader field that describes the segmentation/re-assembly fields required to sucessfully concatenate multiple short messages.
03     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
04     *                        In order to help with reconcilliation of messages
05     * @param unicode This flag is set to true if the message needs to be submitted as a unicode message. This would be for scenario's where the message contains text that does not
06     *                fit within the Latin GSM alphabet. Examples would be messages to be sent in non-western scripts, such as Arabic, Kanji, Chinese, etc.
07     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
08     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
09     *                             Nexmo REST account that represents the outcome of this message.
10     * @param wapPushUrl This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
11     * @param wapPushTitle This is the title that will be associated with the url being submitted to the handset
12     * @param wapPushValidity This is the length of time (in seconds) that the message will be available for once delivered to the handset.
13     *                        Once this time has expired, the url will no longer be visible on the handset to be browsed (Subject to handset compatibility)
14     * @param messageClass (Optional) The message class that is to be applied to this message.
15     * @param protocolId The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
16     */
17    protected Message(final int type,
18                      final String from,
19                      final String to,
20                      final String messageBody,
21                      final byte[] binaryMessageBody,
22                      final byte[] binaryMessageUdh,
23                      final String clientReference,
24                      final boolean unicode,
25                      final boolean statusReportRequired,
26                      final String wapPushUrl,
27                      final String wapPushTitle,
28                      final int wapPushValidity,
29                      final MessageClass messageClass,
30                      final Integer protocolId) {
31        this.type = type;
32
33        this.from = from;
34        this.to = to;
35        this.messageBody = messageBody;
36        this.binaryMessageBody = binaryMessageBody;
37        this.binaryMessageUdh = binaryMessageUdh;
38        this.clientReference = clientReference;
39        this.unicode = unicode;
40        this.statusReportRequired = statusReportRequired;
41        this.wapPushUrl = wapPushUrl;
42        this.wapPushTitle = wapPushTitle;
43        this.wapPushValidity = wapPushValidity;
44        this.messageClass = messageClass;
45        this.protocolId = protocolId;
46    }

48    /**
49     * @return int the type of message will influence the makeup of the request we post to the Nexmo server, and also the action taken by the Nexmo server in response to this message
50     */
51    public int getType() {
52        return this.type;
53    }
54
55    /**
56     * @return String the 'from' address that will be seen on the handset when this message arrives,
57     *                typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
58     */
59    public String getFrom() {
60        return this.from;
61    }
62
63    /**
64     * @return String the phone number of the handset that you wish to send the message to
65     */
66    public String getTo() {
67        return this.to;
68    }
69
70    /**
71     * @return String The text of the message to be sent to the handset
72     */
73    public String getMessageBody() {
74        return this.messageBody;
75    }
76
77    /**
78     * @return byte[] The raw binary message data to be sent to a handset.
79     *                This api, and the Nexmo messaging service will send this data 'as-is' (in conjunction with the binary UDH) and will not make any corrections.
80     *                so you should ensure that it is a correctly constructed message
81     */
82    public byte[] getBinaryMessageBody() {
83        return this.binaryMessageBody == null ? null : this.binaryMessageBody.clone();
84    }
85
86    /**
87     * @return byte[] Most binary content will require a UserDataHeader portion of the message containing commands to enable the handset to interpret the binary data
88     *                (for example, a binary ringtone, a wap-push, OverTheAir configuration, etc).
89     *                Additionally, if you are sending a long text message as multiple concatenated messages and are performing this operation manually rather than
90     *                using the automated long messaging handling in the Nexmo messaging service, then you will need to construct and include here an appropriate
91     *                UserDataHeader field that describes the segmentation/re-assembly fields required to sucessfully concatenate multiple short messages.
92     */
93    public byte[] getBinaryMessageUdh() {
94        return this.binaryMessageUdh == null ? null : this.binaryMessageUdh.clone();
95    }
96
97    /**
98     * @return String This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
99     *                In order to help with reconcilliation of messages
00     */
01    public String getClientReference() {
02        return this.clientReference;
03    }
04
05    /**
06     * @return boolean This flag is set to true if the message needs to be submitted as a unicode message. This would be for scenario's where the message contains text that does not
07     *                 fit within the Latin GSM alphabet. Examples would be messages to be sent in non-western scripts, such as Arabic, Kanji, Chinese, etc.
08     */
09    public boolean isUnicode() {
10        return this.unicode;
11    }
12
13    /**
14     * @return boolean If set to true, then a delivery notification will be requested for this message delivery attempt.
15     *                 Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
16     *                 Nexmo REST account that represents the outcome of this message.
17     */
18    public boolean getStatusReportRequired() {
19        return this.statusReportRequired;
20    }
21
22    /**
23     * @return String This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
24     */
25    public String getWapPushUrl() {
26        return this.wapPushUrl;
27    }
28
29    /**
30     * @return String This is the title that will be associated with the url being submitted to the handset
31     */
32    public String getWapPushTitle() {
33        return this.wapPushTitle;
34    }
35
36    /**
37     * @return int This is the length of time (in seconds) that the message will be available for once delivered to the handset.
38     *             Once this time has expired, the url will no longer be visible on the handset to be browsed (Subject to handset compatibility)
39     */
40    public int getWapPushValidity() {
41        return this.wapPushValidity;
42    }
43
44    /**
45     * @return com.nexmo.messaging.sdk.messages.parameters.MessageClass The message class that is to be applied to this message.
46     */
47    public MessageClass getMessageClass() {
48        return this.messageClass;
49    }
50
51    /**
52     * @return Integer The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
53     */
54    public Integer getProtocolId() {
55        return this.protocolId;
56    }
57
58}
