24import com.nexmo.messaging.sdk.messages.parameters.MessageClass;
25
26/**
27 * TextMessage.java<br><br>
28 *
29 * Represents the details of a plain-text message that is to be submitted via the Nexmo REST api<br><br>
30 *
31 * Created on 5 January 2011, 17:34
32 *
33 * @author  Paul Cook
34 * @version 1.0
35 */
36public class TextMessage extends Message {
37
38    static final long serialVersionUID = 6258872793039443129L;
39
40    /**
41     * Instanciate a new text-message request.<br>
42     * This message will be submitted as a regular 8 bit text message
43     *
44     * @param from the 'from' address that will be seen on the handset when this message arrives,
45     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
46     * @param to   the phone number of the handset that you wish to send the message to
47     * @param messageBody The text of the message to be sent to the handset
48     */
49    public TextMessage(final String from,
50                       final String to,
51                       final String messageBody) {
52        super(MESSAGE_TYPE_TEXT,
53              from,
54              to,
55              messageBody,
56              null,
57              null,
58              null,
59              false,
60              false,
61              null,
62              null,
63              0,
64              null,
65              null);
66    }
67
68    /**
69     * Instanciate a new text-message request, exposing all of the available parameters, and optionally requesting a delivery notification.<br>
70     * This message will be submitted as a regular 8 bit text message
71     *
72     * @param from the 'from' address that will be seen on the handset when this message arrives,
73     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
74     * @param to   the phone number of the handset that you wish to send the message to
75     * @param messageBody The text of the message to be sent to the handset
76     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
77     *                        In order to help with reconcilliation of messages
78     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
79     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
80     *                             Nexmo REST account that represents the outcome of this message.
81     * @param messageClass (Optional) The message class that is to be applied to this message.
82     * @param protocolId The value of the GSM Protocol ID field to be submitted with this message. Ordinarily this should be left as the default value of 0
83     */
84    public TextMessage(final String from,
85                       final String to,
86                       final String messageBody,
87                       final String clientReference,
88                       final boolean statusReportRequired,
89                       final MessageClass messageClass,
90                       final Integer protocolId) {
91        super(MESSAGE_TYPE_TEXT,
92              from,
93              to,
94              messageBody,
95              null,
96              null,
97              clientReference,
98              false,
99              statusReportRequired,
00              null,
01              null,
02              0,
03              messageClass,
04              protocolId);
05    }
06
07}
