24/**
25 * WapPushMessage.java<br><br>
26 *
27 * Represents the details of a wap-push message that is to be submitted via the Nexmo REST api<br><br>
28 *
29 * Created on 5 January 2011, 17:34
30 *
31 * @author  Paul Cook
32 * @version 1.0
33 */
34public class WapPushMessage extends Message {
35
36    static final long serialVersionUID = -8412926886570923963L;
37
38    /**
39     * Instanciate a new wap-push message request, to submit a browsable / downloadable URL to the handset
40     *
41     * @param from the 'from' address that will be seen on the handset when this message arrives,
42     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
43     * @param to   the phone number of the handset that you wish to send the message to
44     * @param wapPushUrl This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
45     * @param wapPushTitle This is the title that will be associated with the url being submitted to the handset
46     */
47    public WapPushMessage(final String from,
48                          final String to,
49                          final String wapPushUrl,
50                          final String wapPushTitle) {
51        super(MESSAGE_TYPE_WAPPUSH,
52              from,
53              to,
54              null,
55              null,
56              null,
57              null,
58              false,
59              false,
60              wapPushUrl,
61              wapPushTitle,
62              0,
63              null,
64              null);
65    }
66
67    /**
68     * Instanciate a new wap-push message request, to submit a browsable / downloadable URL to the handset,
69     * attaching an optional client reference, and optionally requesting a delivery notification
70     *
71     * @param from the 'from' address that will be seen on the handset when this message arrives,
72     *             typically either a valid short-code / long code that can be replied to, or a short text description of the application sending the message (Max 11 chars)
73     * @param to   the phone number of the handset that you wish to send the message to
74     * @param clientReference This is a user definable reference that will be stored in the Nexmo messaging records. It will be available in detailed reporting / analytics
75     *                        In order to help with reconcilliation of messages
76     * @param statusReportRequired If set to true, then a delivery notification will be requested for this message delivery attempt.
77     *                             Upon receiving notification of delivery or failure from the network, the Nexmo platform will submit a notification to the url configured in your
78     *                             Nexmo REST account that represents the outcome of this message.
79     * @param wapPushUrl This is the url that will be submitted to the handset and will appear as a browsable message in the Inbox.
80     * @param wapPushTitle This is the title that will be associated with the url being submitted to the handset
81     * @param wapPushValidity This is the length of time (in seconds) that the message will be available for once delivered to the handset.
82     *                        Once this time has expired, the url will no longer be visible on the handset to be browsed (Subject to handset compatibility)
83     */
84    public WapPushMessage(final String from,
85                          final String to,
86                          final String clientReference,
87                          final boolean statusReportRequired,
88                          final String wapPushUrl,
89                          final String wapPushTitle,
90                          final int wapPushValidity) {
91        super(MESSAGE_TYPE_WAPPUSH,
92              from,
93              to,
94              null,
95              null,
96              null,
97              clientReference,
98              false,
99              statusReportRequired,
00              wapPushUrl,
01              wapPushTitle,
02              wapPushValidity,
03              null,
04              null);
05    }
06
07}
