24import java.util.List;
25import java.util.Map;
26import java.util.TreeMap;
27
28import org.apache.commons.logging.Log;
29import org.apache.commons.logging.LogFactory;
30
31import org.apache.http.NameValuePair;
32import org.apache.http.message.BasicNameValuePair;
33
34import com.nexmo.common.util.MD5Util;
35
36/**
37 * RequestSigning
38 *
39 * Helper class for generating signatures when signing REST requests for submission to Nexmo.
40 * Generates a signature based on the md5 hash of all of the request parameters,
41 * including a timestamp parameter to protect against replay attacks.
42 * The signature is generated using a pre-shared secret key that does not form part of the request.
43 *
44 * @author  Paul Cook
45 * @version
46 */
47public class RequestSigning implements SecurityConstants {
48
49    private static Log log = LogFactory.getLog(RequestSigning.class);
50
51    /**
52     * sign a set of request parameters, generating additional parameters to represent the timestamp and generated signature
53     * uses the supplied pre-shared secret key to generate the signature
54     *
55     * @param params List of NameValuePair instances containing the query parameters for the request that is to be signed
56     * @param secretKey the pre-shared secret key held by the client
57     *
58     * @return String the fully constructed url complete with signature
59     */
60    public static void constructSignatureForRequestParameters(List<NameValuePair> params, String secretKey) {
61        // First, inject a 'timestamp=' parameter containing the current time in seconds since Jan 1st 1970
62        params.add(new BasicNameValuePair(PARAM_TIMESTAMP, "" + System.currentTimeMillis() / 1000));
63
64        Map<String, String> sortedParams = new TreeMap<String, String>();
65        for (NameValuePair param: params) {
66            String name = param.getName();
67            String value = param.getValue();
68            if (name.equals(PARAM_SIGNATURE))
69                continue;
70            if (value == null)
71                value = "";
72            if (!value.trim().equals(""))
73                sortedParams.put(name, value);
74        }
75
76        // Now, walk through the sorted list of parameters and construct a string
77        StringBuilder sb = new StringBuilder();
78        for (Map.Entry<String, String> param: sortedParams.entrySet()) {
79            String name = param.getKey();
80            String value = param.getValue();
81            sb.append("&").append(clean(name)).append("=").append(clean(value));
82        }
83
84        // Now, append the secret key, and calculate an MD5 signature of the resultant string
85        sb.append(secretKey);
86
87        String str = sb.toString();
88
89        String md5 = "no signature";
90        try {
91            md5 = MD5Util.calculateMd5(str);
92        } catch (Exception e) {
93            log.error("error...", e);
94        }
95
96        log.debug("SECURITY-KEY-GENERATION -- String [ " + str + " ] Signature [ " + md5 + " ] ");
97
98        params.add(new BasicNameValuePair(PARAM_SIGNATURE, md5));
99    }
00
01    public static String clean(String str) {
02        return str == null ? null : str.replaceAll("[=&]", "_");
03    }
04
05}
