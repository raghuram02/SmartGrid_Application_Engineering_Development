28    /**
29     * Instanciate a new NexmoSnsClient instance that will communicate using the supplied credentials, and will use the supplied connection and read timeout values.<br>
30     * Additionally, you can specify an alternative service base url. For example submitting to a testing sandbox environment,
31     * or if requested to submit to an alternative address by Nexmo, for example, in cases where it may be necessary to prioritize your traffic.
32     *
33     * @param apiKey Your Nexmo account api key
34     * @param apiSecret Your Nexmo account api secret
35     * @param connectionTimeout over-ride the default connection timeout with this value (in milliseconds)
36     * @param soTimeout over-ride the default read-timeout with this value (in milliseconds)
37     */
38    public NexmoSnsClient(String baseUrl,
39                          final String apiKey,
40                          final String apiSecret,
41                          final int connectionTimeout,
42                          final int soTimeout) throws Exception {
43
44        // Derive a http and a https version of the supplied base url
45        baseUrl = baseUrl.trim();
46        String lc = baseUrl.toLowerCase();
47        if (!lc.startsWith("http://") && !lc.startsWith("https://"))
48            throw new Exception("base url does not start with http:// or https://");
49        if (lc.startsWith("http://")) {
50            this.baseUrlHttp = baseUrl;
51            this.baseUrlHttps = "https://" + baseUrl.substring(7);
52        } else {
53            this.baseUrlHttps = baseUrl;
54            this.baseUrlHttp = "http://" + baseUrl.substring(8);
55        }
56
57        this.apiKey = apiKey;
58        this.apiSecret = apiSecret;
59        this.connectionTimeout = connectionTimeout;
60        this.soTimeout = soTimeout;
61        try {
62            this.documentBuilderFactory = DocumentBuilderFactory.newInstance();
63            this.documentBuilder = this.documentBuilderFactory.newDocumentBuilder();
64        } catch (javax.xml.parsers.ParserConfigurationException e) {
65            throw new Exception("ERROR initializing XML Document builder!", e);
66        }
67    }

79    public SnsServiceResult submit(Request request, boolean https) throws Exception {
80
81        log.info("NEXMO-REST-SNS-SERVICE-CLIENT ... submit request [ " + request.toString() + " ] ");
82
83        // Construct a query string as a list of NameValuePairs
84
85        List<NameValuePair> params = new ArrayList<NameValuePair>();
86
87        params.add(new BasicNameValuePair("api_key", this.apiKey));
88        params.add(new BasicNameValuePair("api_secret", this.apiSecret));
89        params.add(new BasicNameValuePair("cmd", request.getCommand()));
90        if (request.getQueryParameters() != null)
91            for (Map.Entry<String, String> entry: request.getQueryParameters().entrySet())
92                params.add(new BasicNameValuePair(entry.getKey(), entry.getValue()));
93
94        String baseUrl = https ? this.baseUrlHttps : this.baseUrlHttp;
95
96        // Now that we have generated a query string, we can instanciate a HttpClient,
97        // construct a POST or GET method and execute to submit the request
98        String response = null;
99        HttpPost method = new HttpPost(baseUrl);
00        method.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
01        String url = baseUrl + "?" + URLEncodedUtils.format(params, "utf-8");
02        try {
03            if (this.httpClient == null)
04                this.httpClient = HttpClientUtils.getInstance(this.connectionTimeout, this.soTimeout).getNewHttpClient();
05            HttpResponse httpResponse = this.httpClient.execute(method);
06            int status = httpResponse.getStatusLine().getStatusCode();
07            if (status != 200)
08                throw new Exception("got a non-200 response [ " + status + " ] from Nexmo-HTTP for url [ " + url + " ] ");
09            response = new BasicResponseHandler().handleResponse(httpResponse);
10            log.info(".. SUBMITTED NEXMO-HTTP URL [ " + url + " ] -- response [ " + response + " ] ");
11        } catch (Exception e) {
12            log.info("communication failure: " + e);
13            method.abort();
14            return new SnsServiceResult(request.getCommand(),
15                                        SnsServiceResult.STATUS_COMMS_FAILURE,
16                                        "Failed to communicate with NEXMO-HTTP url [ " + url + " ] ..." + e,
17                                        null,
18                                        null);
19        }
20
21        // parse the response doc ...
22
23        /*
24            We receive a response from the api that looks like this, parse the document
25            and turn it into an instance of SnsServiceResult
26
27                <nexmo-sns>
28                    <command>subscribe|publish</command>
29                    <resultCode>0</resultCode>
30                    <resultMessage>OK!</resultMessage>
31                    <transactionId>${transaction-id}</transactionId>
32                    <subscriberArn>${subscriber}</subscriberArn>
33                </nexmo-sns>
34
35        */
36
37        Document doc = null;
38        synchronized(this.documentBuilder) {
39            try {
40                doc = this.documentBuilder.parse(new InputSource(new StringReader(response)));
41            } catch (Exception e) {
42                throw new Exception("Failed to build a DOM doc for the xml document [ " + response + " ] ", e);
43            }
44        }
45
46        String command = null;
47        int resultCode = -1;
48        String resultMessage = null;
49        String transactionId = null;
50        String subscriberArn = null;
51
52        NodeList replies = doc.getElementsByTagName("nexmo-sns");
53        for (int i=0;i<replies.getLength();i++) {
54            Node reply = replies.item(i);
55            NodeList nodes = reply.getChildNodes();
56
57            for (int i2=0;i2<nodes.getLength();i2++) {
58                Node node = nodes.item(i2);
59                if (node.getNodeType() != Node.ELEMENT_NODE)
60                    continue;
61                if (node.getNodeName().equals("command")) {
62                    command = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
63                } else if (node.getNodeName().equals("resultCode")) {
64                    String str = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
65                    try {
66                        resultCode = Integer.parseInt(str);
67                    } catch (Exception e) {
68                        log.error("xml parser .. invalid value in <resultCode> node [ " + str + " ] ");
69                        resultCode = SnsServiceResult.STATUS_INTERNAL_ERROR;
70                    }
71                } else if (node.getNodeName().equals("resultMessage")) {
72                    resultMessage = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
73                } else if (node.getNodeName().equals("transactionId")) {
74                    transactionId = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
75                } else if (node.getNodeName().equals("subscriberArn")) {
76                    subscriberArn = node.getFirstChild() == null ? null : node.getFirstChild().getNodeValue();
77                } else
78                    log.error("xml parser .. unknown node found in nexmo-sns [ " + node.getNodeName() + " ] ");
79            }
80        }
81
82        if (resultCode == -1)
83            throw new Exception("Xml Parser - did not find a <resultCode> node");
84
85        return new SnsServiceResult(command,
86                                    resultCode,
87                                    resultMessage,
88                                    transactionId,
89                                    subscriberArn);
90    }
