24import com.nexmo.sns.sdk.NexmoSnsClient;
25import com.nexmo.sns.sdk.request.SubscribeRequest;
26import com.nexmo.sns.sdk.request.PublishRequest;
27import com.nexmo.sns.sdk.response.PublishResult;
28import com.nexmo.sns.sdk.response.SubscribeResult;
29
30/**
31 * SubscribeAndPublishExample.java<br><br>
32 *
33 * An example of using the nexmo sns api to subscribe a user, then broadcast a message ...<br><br>
34 *
35 * Created on 5 January 2011, 17:34
36 *
37 * @author  Paul Cook
38 * @version 1.0
39 */
40public class SubscribeAndPublishExample {
41
42    public static final String API_KEY = "your-api-key-goes-here";
43    public static final String API_SECRET = "your-api-secret-goes-here";
44
45    public static final String SMS_FROM = "12345";
46    public static final String SMS_TO = "447777111222";
47    public static final String SMS_TEXT = "Hello World!";
48
49    // -- TODO -- fill in your own topic ARN from your amazon SNS console
50    public static final String TOPIC_ARN = "arn:aws:sns:us-east-1:475338436304:paul";
51
52    public static void main(String[] args) {
53
54        // Create a client for submitting to Nexmo
55
56        NexmoSnsClient client = null;
57        try {
58            client = new NexmoSnsClient(API_KEY, API_SECRET);
59        } catch (Exception e) {
60            System.err.println("Failed to instanciate a Nexmo Client");
61            e.printStackTrace();
62            throw new RuntimeException("Failed to instanciate a Nexmo Client");
63        }
64
65        // Create a Request to subscribe a phone number to the SNS service
66
67        SubscribeRequest subscribeRequest = null;
68        try {
69            subscribeRequest = new SubscribeRequest(TOPIC_ARN, SMS_TO);
70        } catch (Exception e) {
71            System.err.println("Failed to construct a SubscriptionRequest");
72            e.printStackTrace();
73            throw new RuntimeException("Failed to construct a SubscriptionRequest");
74        }
75
76        // Use the Nexmo client to submit this subscription
77
78        SubscribeResult subscribeResult = null;
79        try {
80            subscribeResult = client.submit(subscribeRequest);
81        } catch (Exception e) {
82            System.err.println("Failed to perform subscription");
83            e.printStackTrace();
84            throw new RuntimeException("Failed to perform subscription");
85        }
86
87        // Evaluate the subscription response .....
88
89        System.out.println("RESULT OF SUBSCRIPTION REQUEST ....");
90        System.out.println("Command: " + subscribeResult.getCommand());
91        System.out.println("Result: " + subscribeResult.getResultCode() + " - " + subscribeResult.getResultMessage());
92        System.out.println("Subscriber ARN: " + subscribeResult.getSubscriberArn());
93
94        // Create a request to publish a message to all subscribers
95
96        PublishRequest publishRequest = null;
97        try {
98            publishRequest = new PublishRequest(TOPIC_ARN, SMS_FROM, SMS_TEXT);
99        } catch (Exception e) {
00            System.err.println("Failed to construct a PublishRequest");
01            e.printStackTrace();
02            throw new RuntimeException("Failed to construct a PublishRequest");
03        }
04
05        // Use the Nexmo client to submit this subscription
06
07        PublishResult publishResult = null;
08        try {
09            publishResult = client.submit(publishRequest);
10        } catch (Exception e) {
11            System.err.println("Failed to perform publish");
12            e.printStackTrace();
13            throw new RuntimeException("Failed to perform publish");
14        }
15
16        // Evaluate the publish response .....
17
18        System.out.println("RESULT OF SUBSCRIPTION REQUEST ....");
19        System.out.println("Command: " + publishResult.getCommand());
20        System.out.println("Result: " + publishResult.getResultCode() + " - " + publishResult.getResultMessage());
21        System.out.println("Transaction ID: " + publishResult.getTransactionId());
22    }
23
24}
