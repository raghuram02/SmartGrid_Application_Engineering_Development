25 * SnsServiceResult.java<br><br>

29 * Represents the result of the service request to the Nexmo SNS Service
30 *
31 * @author  Paul Cook
32 * @version 1.0
33 */
34public class SnsServiceResult implements PublishResult,
35                                         SubscribeResult {
36
37    public static final int STATUS_OK = 0;
38    public static final int STATUS_BAD_COMMAND = 1;
39    public static final int STATUS_INTERNAL_ERROR = 2;
40    public static final int STATUS_INVALID_ACCOUNT = 3;
41    public static final int STATUS_MISSING_TOPIC = 4;
42    public static final int STATUS_INVALID_OR_MISSING_MSISDN = 5;
43    public static final int STATUS_INVALID_OR_MISSING_FROM = 6;
44    public static final int STATUS_INVALID_OR_MISSING_MSG = 7;
45    public static final int STATUS_TOPIC_NOT_FOUND = 8;
46    public static final int STATUS_TOPIC_PERMISSION_FAILURE = 9;
47    public static final int STATUS_COMMS_FAILURE = 13;
48
49    private final String command;
50    private final int resultCode;
51    private final String resultMessage;
52    private final String transactionId;
53    private final String subscriberArn;
54
55    public SnsServiceResult(final String command,
56                            final int resultCode,
57                            final String resultMessage,
58                            final String transactionId,
59                            final String subscriberArn) {
60        this.command = command;
61        this.resultCode = resultCode;
62        this.resultMessage = resultMessage;
63        this.transactionId = transactionId;
64        this.subscriberArn = subscriberArn;
65    }
66
67    @Override
68    public String getCommand() {
69        return this.command;
70    }
71
72    @Override
73    public int getResultCode() {
74        return this.resultCode;
75    }
76
77    @Override
78    public String getResultMessage() {
79        return this.resultMessage;
80    }
81
82    @Override
83    public String getTransactionId() {
84        return this.transactionId;
85    }
86
87    @Override
88    public String getSubscriberArn() {
89        return this.subscriberArn;
90    }
91
92}
